//各表的创建
-- 创建 condition 表
CREATE TABLE condition (
    student_id VARCHAR(20) PRIMARY KEY, -- 学号
    name VARCHAR(50) NOT NULL,          -- 姓名
    graduation_year INT NOT NULL,       -- 毕业年份
    gender VARCHAR(10) NOT NULL,        -- 性别
    political_status VARCHAR(50),       -- 政治面貌
    major VARCHAR(100) NOT NULL,        -- 专业
    hardship_category VARCHAR(50),      -- 困难生类别
    address VARCHAR(255)                -- 地区
);

-- 创建 second_course 表
CREATE TABLE second_course (
    student_id VARCHAR(20) PRIMARY KEY, -- 学号
    courses_count INT NOT NULL          -- 参与课程门数
);

-- 创建 award 表
CREATE TABLE award (
    student_id VARCHAR(20) PRIMARY KEY, -- 学号
    scholarships_count INT NOT NULL     -- 获得奖学金次数
);

-- 创建 job 表
CREATE TABLE job (
    student_id VARCHAR(20) PRIMARY KEY,      -- 学号
    graduation_destination VARCHAR(100),    -- 毕业去向
    organization_name VARCHAR(255),         -- 单位名称
    organization_type VARCHAR(100),         -- 单位性质
    industry VARCHAR(100),                  -- 单位行业
    location VARCHAR(255),                  -- 单位所在地
    position_category VARCHAR(100)          -- 工作职位类别
);
